'id' - индентификатор записи

'from_id' - id автора записи

'owner_id' - id страницы, на которой запись опубликована

'date' - дата в секундах с 1970 года до выкладывания записи

'post_type' - тип поста

'text' - текст, который там написан

'attachments' - вложения, которые использовались

'post_source' - источник поста

'comments' - комментарии

'likes' - отметки "нравится"

'reposts' - кто поделился записью
